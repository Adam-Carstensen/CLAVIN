39      //this convenience method turns an array of location name strings into a list of occurrences with fake positions.
40      //(useful for tests that don't care about position in the document)
41      public static ExtractionContext makeOccurrencesFromNames(String[] locationNames) {
42          List<LocationOccurrence> locations = new ArrayList<LocationOccurrence>(locationNames.length);
43          for(int i = 0; i < locationNames.length; ++i ) {
44              locations.add(new LocationOccurrence(locationNames[i], i));
45          }
46          return new ExtractionContext("", locations, null);
47      }

59  		
60  		assertEquals("LocationResolver failed exact String match", RESTON_VA, resolvedLocations.get(0).getPlace().getId());
61  		assertEquals("LocationResolver failed on all lowercase", RESTON_VA, resolvedLocations.get(1).getPlace().getId());
62  		assertEquals("LocationResolver failed on all uppercase", RESTON_VA, resolvedLocations.get(2).getPlace().getId());
63  		assertEquals("LocationResolver failed on alternate name", RESTON_VA, resolvedLocations.get(3).getPlace().getId());
