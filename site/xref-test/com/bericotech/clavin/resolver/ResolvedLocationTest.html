40  /**
41   * Basic tests for the {@link ResolvedLocation} class, which
42   * encapsulates a {@link GeoName} object representing the best match
43   * between a given location name and gazetter record, along with some
44   * information about the geographic entity resolution process.
45   *
46   */
47  public class ResolvedLocationTest {
48  
49      /**
50       * Ensures proper performance of the overridden equals() method.
51       */
52      @Test
53      public void testEquals() {
54          // two identical sample gazetteer records from GeoNames.org
55          String geonamesEntry = "4781530\tReston\tReston\tReston,Рестон\t38.96872\t-77.3411\tP\tPPL\tUS\tVA\t059\t58404\t100\t102\tAmerica/New_York\t2011-05-14";
56          String geonamesEntry2 = "478153\tReston\tReston\tReston,Рестон\t38.96872\t-77.3411\tP\tPPL\tUS\tVA\t059\t58404\t100\t102\tAmerica/New_York\t2011-05-14";
57          
58          // create corresponding Lucene Documents for gazetteer records
59          Document luceneDoc = IndexDirectoryBuilder.buildDoc("Nowhere", geonamesEntry, 999, (long)999);
60          Document luceneDoc2 = IndexDirectoryBuilder.buildDoc("Nowhere", geonamesEntry2, 222, (long)999);
61          
62          // a bogus LocationOccurrence object for testing
63          LocationOccurrence locationA = new LocationOccurrence("A", 0);
64              
65          // two ResolvedLocation objects created from same Lucene Doc, etc.
66          ResolvedLocation resolvedLocation = new ResolvedLocation(luceneDoc, locationA, false);
67          ResolvedLocation resolvedLocationDupe = new ResolvedLocation(luceneDoc, locationA, false);
68          
69          // an identical ResolvedLocation object created from the second Lucene doc
70          ResolvedLocation resolvedLocation2 = new ResolvedLocation(luceneDoc2, locationA, false);
71          
72          assertTrue("ResolvedLocation == self", resolvedLocation.equals(resolvedLocation));
73          assertFalse("ResolvedLocation =! null", resolvedLocation.equals(null));
74          assertFalse("ResolvedLocation =! different class", resolvedLocation.equals(new Integer(0)));
75          assertTrue("ResolvedLocation == dupe", resolvedLocation.equals(resolvedLocationDupe));
76          assertFalse("ResolvedLocation != different geonameID", resolvedLocation.equals(resolvedLocation2));
77      }
