39  /**
40   * Checks output produced by {@link GeoParser}, which is more of an
41   * integration test than a unit test.
42   *
43   */
44  public class GeoParserTest {
45      
46      // expected geonameID numbers for given location names
47      int UNITED_STATES = 6252001;
48      int VERMONT = 5242283;
49      int MASSACHUSETTS = 6254926;
50      
51      /**
52       * Ensures we're getting good output from the end-to-end GeoParser
53       * process.
54       * @throws Exception
55       */
56      @Test
57      public void testParse() throws Exception {
58          // instantiate the CLAVIN GeoParser
59          GeoParser parser = GeoParserFactory.getDefault("./IndexDirectory");
60          
61          // sample text to be geoparsed
62          String inputText = "Calvin Coolidge was the 30th president " +
63                  "of the United States. He was born in Vermont and " +
64                  "died in Massachusetts.";
65          
66          // parse location names in the text into geographic entities
67          List<ResolvedLocation> resolvedLocations = parser.parse(inputText);
68          
69          // check the output
70          assertEquals("Wrong number of ResolvedLocations", 3, resolvedLocations.size());
71          assertEquals("Incorrect ResolvedLocation", UNITED_STATES, resolvedLocations.get(0).geoname.geonameID);
72          assertEquals("Incorrect ResolvedLocation", VERMONT, resolvedLocations.get(1).geoname.geonameID);
73          assertEquals("Incorrect ResolvedLocation", MASSACHUSETTS, resolvedLocations.get(2).geoname.geonameID);
74          assertEquals("Incorrect position of LocationOccurance", inputText.indexOf("United States"), resolvedLocations.get(0).location.position);
75          assertEquals("Incorrect position of LocationOccurance", inputText.indexOf("Vermont"), resolvedLocations.get(1).location.position);
76          assertEquals("Incorrect position of LocationOccurance", inputText.indexOf("Massachusetts"), resolvedLocations.get(2).location.position);
77  
78      }
