58  /**
59   * Builds a Lucene index of geographic entries based on
60   * the GeoNames gazetteer.
61   * 
62   * This program is run one-time before CLAVIN can be used.
63   * 
64   */
65  public class IndexDirectoryBuilder {
66      
67      public final static Logger logger = LoggerFactory.getLogger(IndexDirectoryBuilder.class);
68      
69      // the GeoNames gazetteer file to be loaded
70      static String pathToGazetteer = "./allCountries.txt";
71  
72      /**
73       * Turns a GeoNames gazetteer file into a Lucene index, and adds
74       * some supplementary gazetteer records at the end.
75       * 
76       * @param args              not used
77       * @throws IOException
78       */
79      public static void main(String[] args) throws IOException {
80          
81          // exit gracefully if the directory exists 
82          File idir = new File("./IndexDirectory");
83          if (idir.exists() ) {
84              logger.info("IndexDirectory exists. Remove the directory and try again.");
85              System.exit(-1);
86          }
87          
88          logger.info("Indexing... please wait.");
89          
90          // Create a new index file on disk, allowing Lucene to choose
91          // the best FSDirectory implementation given the environment.
92          FSDirectory index = FSDirectory.open(idir);
93          
94          // indexing by lower-casing & tokenizing on whitespace
95          Analyzer indexAnalyzer = new WhitespaceLowerCaseAnalyzer();
96          
97          // create the object that will actually build the Lucene index
98          IndexWriter indexWriter = new IndexWriter(index, new IndexWriterConfig(Version.LUCENE_40, indexAnalyzer));
99          
100         // open the gazetteer files to be loaded
101         BufferedReader r = new BufferedReader(new InputStreamReader(new FileInputStream(new File(pathToGazetteer)), "UTF-8"));
102         BufferedReader r2 = new BufferedReader(new InputStreamReader(new FileInputStream(new File("./src/main/resources/SupplementaryGazetteer.txt")), "UTF-8"));
103         
104         String line;
105         
106         // let's see how long this takes...
107         Date start = new Date();
108         
109         // load GeoNames gazetteer into Lucene index
110         int count = 0;
111         while ((line = r.readLine()) != null)
112             try {
113                 count += 1;
114                 // print progress update to console
115                 if (count % 100000 == 0 ) logger.info("rowcount: " + count);
116                 addToIndex(indexWriter, line);
117             } catch (Exception e) {
118                  logger.info("Skipping... Error on line:" + line);
119             }
120 
121         // add supplementary gazetteer records to index
122         while ((line = r2.readLine()) != null)
123             addToIndex(indexWriter, line);
124         
125         // that wasn't so long, was it?
126         Date stop = new Date();
127         
128         logger.info("[DONE]");
129         logger.info(indexWriter.maxDoc() + " geonames added to index.");
130         logger.info("Merging indices... please wait.");
131         
132         indexWriter.close();
133         index.close();
134         r.close();
135         r2.close();
136         
137         logger.info("[DONE]");
138         
139         DateFormat df = new SimpleDateFormat("HH:mm:ss");
140         long elapsed_MILLIS = stop.getTime() - start.getTime();
141         logger.info("Process started: " + df.format(start) + ", ended: " + df.format(stop)
142                 + "; elapsed time: " + MILLISECONDS.toSeconds(elapsed_MILLIS) + " seconds.");
143     }

145     /**
146      * Adds entries to the Lucene index for each unique name associated
147      * with a {@link GeoName} object.
148      * 
149      * @param indexWriter   the object that actually builds the Lucene index
150      * @param geonameEntry  single record from GeoNames gazetteer
151      * @throws IOException
152      */
153     private static void addToIndex(IndexWriter indexWriter, String geonameEntry) throws IOException {
154         // create a GeoName object from a single gazetteer record
155         GeoName geoname = GeoName.parseFromGeoNamesRecord(geonameEntry);
156         
157         // add the primary (UTF-8) name for this location
158         if (geoname.name.length() > 0)
159             indexWriter.addDocument(buildDoc(geoname.name, geonameEntry, geoname.geonameID, geoname.population));
160         
161         // add the ASCII name if it's different from the primary name
162         if (geoname.asciiName.length() > 0 && !geoname.asciiName.equals(geoname.name))
163             indexWriter.addDocument(buildDoc(geoname.asciiName, geonameEntry, geoname.geonameID, geoname.population));
164         
165         // add alternate names (if any) if they differ from the primary
166         // and alternate names
167         for (String altName : geoname.alternateNames)
168             if (altName.length() > 0 && !altName.equals(geoname.name) && !altName.equals(geoname.name))
169                 indexWriter.addDocument(buildDoc(altName, geonameEntry, geoname.geonameID, geoname.population));
170     }

172     /**
173      * Builds a Lucene document to be added to the index based on a
174      * specified name for the location and the corresponding
175      * {@link GeoName} object.
176      * 
177      * @param name          name to serve as index key
178      * @param geonameEntry  string from GeoNames gazetteer
179      * @param geonameID     unique identifier (for quick look-up)
180      * @param population    number of inhabitants (used for scoring)
181      * @return              document to be added to the index
182      */
183     public static Document buildDoc(String name, String geonameEntry, int geonameID, Long population) {
184         // in case you're wondering, yes, this is a non-standard use of
185         // the Lucene Document construct :)
186         Document doc = new Document();
187         
188         // this is essentially the key we'll try to match location
189         // names against
190         doc.add(new TextField("indexName", name, Field.Store.YES));
191         
192         // this is the payload we'll return when matching location
193         // names to gazetteer records
194         doc.add(new StoredField("geoname", geonameEntry));
195         
196         // TODO: use geonameID to link administrative subdivisions to
197         //       each other
198         doc.add(new IntField("geonameID", geonameID, Field.Store.YES));
199         
200         // we'll initially sort match results based on population
201         doc.add(new LongField("population", population, Field.Store.YES));
202         
203         logger.debug("Adding to index: " + name);
204         
205         return doc;
206     }
