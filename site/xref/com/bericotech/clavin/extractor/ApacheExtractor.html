43  /**
44   * Extracts location names from unstructured text documents using a
45   * named entity recognizer (Apache OpenNLP Name Finder).
46   *
47   */
48  public class ApacheExtractor implements LocationExtractor {
49      
50      // the actual named entity recognizer (NER) object
51      private NameFinderME nameFinder;
52      
53      // used to tokenize plain text into the OpenNLP format
54      private TokenizerME tokenizer;
55  
56      // used to split the input into sentences before finding names
57      private SentenceDetectorME sentenceDetector;
58      
59      // resource files used by Apache OpenNLP Name Finder
60      private static final String pathToNERModel = "/en-ner-location.bin";
61      private static final String pathToTokenizerModel = "/en-token.bin";
62      private static final String pathToSentenceDetectorModel = "/en-sent.bin";
63  
64      
65      /**
66       * Builds an {@link ApacheExtractor} by instantiating the OpenNLP
67       * Name Finder and Tokenizer.
68       * 
69       * @throws IOException 
70       */
71      public ApacheExtractor() throws IOException {
72          nameFinder = new NameFinderME(new TokenNameFinderModel(ApacheExtractor.class.getResourceAsStream(pathToNERModel)));
73          tokenizer = new TokenizerME(new TokenizerModel(ApacheExtractor.class.getResourceAsStream(pathToTokenizerModel)));
74          sentenceDetector = new SentenceDetectorME(new SentenceModel(ApacheExtractor.class.getResourceAsStream(pathToSentenceDetectorModel)));
75      }
76      
77      /**
78       * Extracts location names from unstructured text using the named
79       * entity recognizer (NER) feature provided by the Apache OpenNLP
80       * Name Finder.
81       * 
82       * @param plainText     Contents of text document
83       * @return List of location names and positions
84       */
85      public List<LocationOccurrence> extractLocationNames(String plainText) {
86          if(plainText == null) {
87              throw new IllegalArgumentException("plaintext input to extractLocationNames should not be null");
88          }
89  
90          List<LocationOccurrence> nerResults = new ArrayList<LocationOccurrence>();
91  
92          // The values used in these Spans are string character offsets
93          Span sentenceSpans[] = sentenceDetector.sentPosDetect(plainText);
94  
95          // Each sentence gets processed on its own
96          for (Span sentenceSpan : sentenceSpans) {
97  
98              // find the start and end position of this sentence in the document
99              String sentence = plainText.substring(sentenceSpan.getStart(), sentenceSpan.getEnd());
100 
101             // tokenize the text into the required OpenNLP format
102             String[] tokens = tokenizer.tokenize(sentence);
103 
104             //the values used in these Spans are string character offsets of each token from the sentence beginning
105             Span[] tokenPositionsWithinSentence = tokenizer.tokenizePos(sentence);
106 
107             // find the location names in the tokenized text
108             // the values used in these Spans are NOT string character offsets, they are indices into the 'tokens' array
109             Span names[] = nameFinder.find(tokens);
110 
111 
112             //for each name that got found, create our corresponding occurrence
113             for (Span name : names) {
114 
115                 //find offsets relative to the start of the sentence
116                 int beginningOfFirstWord = tokenPositionsWithinSentence[name.getStart()].getStart();
117                 int endOfLastWord = tokenPositionsWithinSentence[name.getEnd() - 1].getEnd(); //-1 because the high end of a Span is noninclusiv
118 
119                 //to get offsets relative to the document as a whole, just add the offset for the sentence itself
120                 int startOffsetInDoc = sentenceSpan.getStart() + beginningOfFirstWord;
121                 int endOffsetInDoc = sentenceSpan.getStart() + endOfLastWord;
122 
123                 //look back into the original input string to figure out what the text is that I got a hit on
124                 String nameInDocument = plainText.substring(startOffsetInDoc, endOffsetInDoc);
125 
126                 // add to List of results to return
127                 nerResults.add(new LocationOccurrence(nameInDocument, startOffsetInDoc));
128             }
129 
130         }
131 
132         // this is necessary to maintain consistent results across
133         // multiple runs on the same data, which is what we want
134         nameFinder.clearAdaptiveData();
135 
136         return nerResults;
137     }
