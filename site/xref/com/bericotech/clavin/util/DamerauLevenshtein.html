35  /**
36   * Utility functions for calculating edit distance between strings,
37   * consistent with the Damerau-Levenshtein metric.
38   * 
39   * Transposition, substitution, insertion, and deletion operations are
40   * all considered to be one edit each. Multiple substring edits (e.g.,
41   * adjacent transpositions) are supported, unlike in "optimal string
42   * alignment distance" where no substring may be edited more than once.
43   */
44  public class DamerauLevenshtein {
45      // sentinel value for the end of contents in an "infinite" array
46      final static Null endMarker = new Null();
47  
48      /**
49       * Computes the true Damerau–Levenshtein edit distance
50       * (with adjacent transpositions) between two given strings.<br><br>
51       * 
52       * Based on <a href="http://en.wikipedia.org/wiki/Damerau–Levenshtein_distance">C# code from Wikipedia</a>.
53       * 
54       * @param str1  First string being compared
55       * @param str2  Second string being compared
56       * @return      Edit distance between strings
57       */
58      public static int damerauLevenshteinDistance(String str1, String str2) {
59          // return fast if one or both strings is empty or null
60          if ((str1 == null) || str1.isEmpty()) {
61              if ((str2 == null) || str2.isEmpty()) {
62                  return 0;
63              } else {
64                  return str2.length();
65              }
66          } else if ((str2 == null) || str2.isEmpty()) {
67              return str1.length();
68          }
69          
70          // split strings into string arrays
71          String[] stringArray1 = str1.split("");
72          String[] stringArray2 = str2.split("");
73          
74          // initialize matrix values
75          int[][] matrix = new int[stringArray1.length + 2][stringArray2.length + 2];
76          int bound = stringArray1.length + stringArray2.length;
77          matrix[0][0] = bound;
78          for (int i = 0; i <= stringArray1.length; i++) {
79              matrix[i + 1][1] = i;
80              matrix[i + 1][0] = bound;
81          }
82          for (int j = 0; j <= stringArray2.length; j++) {
83              matrix[1][j + 1] = j;
84              matrix[0][j + 1] = bound;
85          }
86          
87          // initialize dictionary
88          SortedMap<String, Integer> dictionary = new TreeMap<String, Integer>();
89          for (String letter : (str1 + str2).split("")) {
90              if (!dictionary.containsKey(letter)) {
91                  dictionary.put(letter, 0);
92              }
93          }
94          
95          // compute edit distance between strings
96          for (int i = 1; i <= stringArray1.length; i++) {
97              int index = 0;
98              for (int j = 1; j <= stringArray2.length; j++) {
99                  int i1 = dictionary.get(stringArray2[j - 1]);
100                 int j1 = index;
101                 if (stringArray1[i - 1].equals(stringArray2[j - 1])) {
102                     matrix[i + 1][j + 1] = matrix[i][j];
103                     index = j;
104                 } else {
105                     matrix[i + 1][j + 1] = Math.min(matrix[i][j], Math.min(matrix[i + 1][j], matrix[i][j + 1])) + 1;
106                 }
107                 
108                 matrix[i + 1][j + 1] = Math.min(matrix[i + 1][j + 1], matrix[i1][j1] + (i - i1 - 1) + 1 + (j - j1 - 1));
109             }
110             
111             dictionary.put(stringArray1[i - 1], i);
112         }
113         
114         return matrix[stringArray1.length + 1][stringArray2.length + 1];
115     }

117     /**
118      * Convenience method for calling {@link #damerauLevenshteinDistance(String str1, String str2)}
119      * when you don't care about case sensitivity.
120      * 
121      * @param str1  First string being compared
122      * @param str2  Second string being compared
123      * @return      Case-insensitive edit distance between strings
124      */
125     public static int damerauLevenshteinDistanceCaseInsensitive(String str1, String str2) {
126         return damerauLevenshteinDistance(str1.toLowerCase(), str2.toLowerCase());
127     }

129     /**
130      * Fast method for determining whether the Damerau-Levenshtein edit
131      * distance between two strings is less than 2.
132      * 
133      * Returns as quick as possibly by stopping once multiple edits are
134      * found. Significantly faster than {@link #damerauLevenshteinDistance(String str1, String str2)}
135      * which explores every path between every string to get the exact
136      * edit distance. Despite the speed boost, we maintain consistency
137      * with {@link #damerauLevenshteinDistance(String str1, String str2)}.
138      * 
139      * @param str1  First string being compared
140      * @param str2  Second string being compared
141      * @return      True if DL edit distance < 2, false otherwise
142      */
143     public static boolean isEditDistance1(String str1, String str2) {
144         // one or both strings is empty or null
145         if ((str1 == null) || str1.isEmpty()) {
146             if ((str2 == null) || str2.isEmpty()) {
147                 return true;
148             } else {
149                 return (str2.length() <= 1);
150             }
151         } else if ((str2 == null) || str2.isEmpty()) {
152             return (str1.length() <= 1);
153         }
154         
155         // difference between string lengths ensures edit distance > bound
156         if (Math.abs(str1.length() - str2.length()) > 1) return false;
157         
158         // initialize counters
159         int offset1 = 0;
160         int offset2 = 0;
161         int i = 0;
162         
163         InfiniteCharArray chars1 = new InfiniteCharArray(str1.toCharArray());
164         InfiniteCharArray chars2 = new InfiniteCharArray(str2.toCharArray());
165         
166         while (!chars1.get(i + offset1).equals(endMarker) || !chars2.get(i + offset2).equals(endMarker)) {
167             if (!chars1.get(i + offset1).equals(chars2.get(i + offset2))) { // character mismatch
168                 if ((chars1.get(i + offset1).equals(chars2.get(i + offset2 + 1))) &&
169                         (chars1.get(i + offset1 + 1).equals(chars2.get(i + offset2))) &&
170                         (chars1.remainder(i + offset1 + 2).equals(chars2.remainder(i + offset2 + 2)))) { // transposition
171                     i = i + 2; // move past the transposition
172                 } else if (chars1.remainder(i + offset1).equals(chars2.remainder(i + offset2 + 1))) { // insertion
173                     offset2++; // realign
174                 } else if (chars1.remainder(i + offset1 + 1).equals(chars2.remainder(i + offset2))) { // deletion
175                     offset1++; // realign
176                 } else if (chars1.remainder(i + offset1 + 1).equals(chars2.remainder(i + offset2 + 1))) { // substitution
177                     i++; // 
178                 } else return false; // multiple edits
179             }
180             
181             i++;
182         }
183         
184         return true;
185     }

188 /**
189  * Convenience class allowing us to avoid running into
190  * ArrayIndexOutOfBoundsExceptions (thanks, Java!)
191  *
192  */
193 class InfiniteCharArray {
194     // the array being encapsulated
195     private char[] array;
196     
197     /**
198      * Sole constructor.
199      * 
200      * @param array     the array to be encapsulated
201      */
202     protected InfiniteCharArray(char[] array) {
203         this.array = array;
204     }
205     
206     /**
207      * If we try to retrieve what lies beyond the end of values, return
208      * a {@link Null} object instead of throwing an
209      * ArrayIndexOutOfBoundsException. Otherwise, return the value at
210      * at the given index.
211      * 
212      * @param index     the position in the array for which we seek a value
213      * @return          the value at that index or a Null object if we're "out of bounds"
214      */
215     protected Object get(int index) {
216         if (index < this.array.length) {
217             return this.array[index];
218         } else {
219             return new Null();
220         }
221     }
222     
223     /**
224      * Get the contents of the char array to the right of the given
225      * index, and return it as a String.
226      * 
227      * @param index     left bound of the string we're pulling from the char array
228      * @return          a string representing everything to the right of the index
229      */
230     protected String remainder(int index) {
231         if (index > this.array.length)
232             return "";
233         else return new String(Arrays.copyOfRange(this.array, index, this.array.length));
234     }
235 }
