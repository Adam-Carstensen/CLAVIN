39   * Really simple, reusable weighting strategy for evaluating items.
40   *
41   * @param <ITEM> Item to weigh
42   * @param <CONTEXT> Context to help in the weighing

44  public class GenericWeightedSelectionStrategy<ITEM, CONTEXT> {

49  	 * Please ensure you add the weighers you would like to use.

55  	 * weighers to use when evaluating ITEM's.

67  	 */
68  	public void setWeighers(Collection<Weigher<ITEM, CONTEXT>> weighers){
69  		
70  		if (weighers != null && weighers.size() > 0){
71  			
72  			this.weighers.addAll(weighers);
73  		}
74  	}

76  	/**
77  	 * Select the best candidate from each list of candidates that represent
78  	 * the best selection for that collection.
79  	 * @param candidatesList List of Candidate Lists
80  	 * @param context Context to help weighers.
81  	 * @param options Options for configuring the weighers
82  	 * @return List of the best selections.
83  	 * @throws Exception
84  	 */
85  	public List<ITEM> select(List<List<ITEM>> candidatesList, CONTEXT context, Options options) 
86  			throws Exception {
87  		
88  		ArrayList<ITEM> bestSelections = new ArrayList<ITEM>();
89  		
90  		for (List<ITEM> candidates : candidatesList){
91  			
92  			ITEM bestSelection = 
93  				makeSelection(candidates, context, options);
94  			
95  			if (bestSelection != null){
96  				
97  				bestSelections.add(bestSelection);
98  			}
99  		}
100 		
101 		return bestSelections;
102 	}

104 	/**
105 	 * Given a list of Candidates, select the best item.
106 	 * @param candidates Candidates to choose from.
107 	 * @param context Context to help coach the weighers.
108 	 * @param options Options to help configure the weighers.
109 	 * @return Best Selection or null.
110 	 */
111 	protected ITEM makeSelection(List<ITEM> candidates, CONTEXT context, Options options){
112 		
113 		ITEM bestSelection = null;
114 		double bestWeight = -1;
115 		
116 		if (candidates.size() > 0){
117 			
118 			for (ITEM candidate : candidates){
119 				
120 				double weight = getWeight(candidate, context, options);
121 				
122 				if (weight > bestWeight){
123 					
124 					bestWeight = weight;
125 					
126 					bestSelection = candidate;
127 					
128 					System.out.println("New Best: " + weight + ", Loc: " + candidate);
129 				}
130 			}
131 		}
132 		
133 		
134 		
135 		return bestSelection;
136 	}

138 	/**
139 	 * Get the overall weight of the candidate ITEM by
140 	 * taking the sum of weights provided by our weighers.
141 	 * @param candidate Candidate item to weigh
142 	 * @param context Addition context for the weighing algorithms.
143 	 * @param options Options to help configure the weighers.
144 	 * @return The numeric score of the selection based on the
145 	 * weight provided by the weighing algorithms.
146 	 */
147 	protected double getWeight(ITEM candidate, CONTEXT context, Options options){
148 		
149 		double weight = 0;
150 		
151 		for (Weigher<ITEM, CONTEXT> weigher : this.weighers){
152 			
153 			weight += weigher.weigh(candidate, context, options);
154 		}
155 		
156 		return weight;
157 	}
