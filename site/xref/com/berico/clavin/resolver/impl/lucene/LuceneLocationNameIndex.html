50  /**
51   * Location Name Index backed by Lucene index.
52   * 
53   * This implementation utilizes two strategies:
54   * 1.  Attempt an exact match on the name provided.
55   * 2.  (if set and exact match returns without results), perform
56   *     a fuzzy match against the index on the name provided.
57   * 
58   * By default, results are sorted first by population, and then by
59   * field score.  This works well for location names that are subsets
60   * of a index's normalized name ("New York" in "City of New York"), but
61   * will definitely exhibit a population bias for results like Boston, MA
62   * and Boston, Philippines.
63   */
64  public class LuceneLocationNameIndex implements LocationNameIndex {
65  
66  	private static final Logger logger = 
67  			LoggerFactory.getLogger(LuceneLocationNameIndex.class);
68  	
69  	/**
70  	 * Default maximum number of results to return (by Lucene).
71  	 */
72  	public static final int DEFAULT_LIMIT = 10;
73  	public static final String KEY_DEFAULT_LIMIT = "location.index.limit";
74  	
75  	/**
76  	 * Whether fuzzy matching should be used by default.
77  	 */
78  	public static final boolean DEFAULT_USE_FUZZY = false;
79  	public static final String KEY_DEFAULT_USE_FUZZY = "location.index.useFuzzy";
80  	
81  	/**
82  	 * Default sorting mechanism (Population, then Field Score).
83  	 * It's important to note that this mechanism favors population size of
84  	 * results.  The assumption is that "Boston" will resolve to "City of Boston"
85  	 * before "Boston" (exact term match for location in Philippines) or something
86  	 * more specific like "Boston Heights".
87  	 */
88  	public static Sort DEFAULT_SORTER = 
89  		new Sort(
90  			new SortField(FieldConstants.POPULATION, SortField.Type.LONG, true),
91  			SortField.FIELD_SCORE);
92  	
93  	LuceneComponents lucene;
94  	AnalyzingQueryParser queryParser;
95  	
96  	/**
97  	 * Instantiate the Index with the appropriate LuceneComponents.
98  	 * @param lucene Configured LuceneComponents.
99  	 */
100 	public LuceneLocationNameIndex(LuceneComponents lucene){
101 		
102 		this.lucene = lucene;
103 		this.queryParser = new AnalyzingQueryParser(
104 			Version.LUCENE_43, FieldConstants.NAME, lucene.getIndexAnalyzer());
105 	}

107 	/**
108 	 * Return a list of Resolved Locations that best match the Location Occurrence
109 	 * found in a document.
110 	 * @param occurrence The Location Occurrence.
111 	 * @param options Options for the index.
112 	 * @return List of Resolved Locations matching the occurrence.
113 	 */
114 	@Override
115 	public List<ResolvedLocation> search(
116 			LocationOccurrence occurrence, Options options) throws Exception {
117 		
118 		options = (options == null)? new Options() : options;
119 		
120 		// Get the max number of records to return.
121 		int limit = options.getInt(KEY_DEFAULT_LIMIT, DEFAULT_LIMIT);
122 		
123 		// Get whether fuzzy matching is enabled.
124 		boolean useFuzzy = options.getBoolean(KEY_DEFAULT_USE_FUZZY, DEFAULT_USE_FUZZY);
125 		
126 		IndexSearcher searcher = lucene.getSearcherManager().acquire();
127 		
128 		boolean usedFuzzy = false;
129 		
130 		// We need to sanitize the name so it doesn't have unescaped Lucene syntax that
131 		// would throw off the search index.
132 		String escapedName = 
133 				QueryParserBase.escape(occurrence.getText().toLowerCase());
134 		
135 		// Try an exact query
136 		Query query = getExactQuery(escapedName);
137 		
138 		// Gather the results.
139 		TopDocs results = searcher.search(query, null, limit, DEFAULT_SORTER);
140 		
141 		// If there are no results, and a fuzzy query was requested
142 		if (results.scoreDocs.length == 0 && useFuzzy) {
143 			
144 			usedFuzzy = true;
145 			
146 			// Attempt a fuzzy query
147 			query = getFuzzyQuery(escapedName);
148 			
149 			// Gather the results
150 			results = searcher.search(query, null, limit, DEFAULT_SORTER);
151 		}
152 		
153 		if (results.scoreDocs.length == 0)
154 			logger.info("Found no results for {}.", escapedName);
155 		
156 		return LuceneUtils.convertToLocations(occurrence, searcher, results, usedFuzzy);
157 	}

159 	/**
160 	 * Construct an exact query for the provided location name.
161 	 * @param locationName Name to search for.
162 	 * @return Exact Query
163 	 * @throws ParseException
164 	 */
165 	protected Query getExactQuery(String locationName) throws ParseException {
166 	
167 		// We want to attempt to force an 'exact match', but using quotes in the 
168 		// search string.  We also want to search in lower case to avoid
169 		// unnormalized names or gramatical errors.
170 		String searchExpression = String.format("\"%s\"", locationName.toLowerCase());
171 		
172 		// Parse the Lucene query
173 		return queryParser.parse(searchExpression);
174 	}
175 	
176 	/**
177 	 * Construct a fuzzy query for the provided location name.
178 	 * @param locationName Name to search for.
179 	 * @return Fuzzy Query
180 	 * @throws ParseException
181 	 */
182 	protected Query getFuzzyQuery(String locationName) throws ParseException{
183 
184 		// Adding a tilde at the end of the query will instruct Lucene to perform
185 		// a fuzzy query.
186 		String searchExpression = String.format("%s~", locationName.toLowerCase());
187 		
188 		// Parse the Lucene query
189 		return queryParser.parse(searchExpression);
190 	}
