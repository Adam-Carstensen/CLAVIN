44  /**
45   * Performs geoparsing of documents; extracts location names from
46   * unstructured text and resolves them against a gazetteer to produce
47   * structured geo data.
48   * 
49   * Main API entry point for CLAVIN -- simply instantiate this class and
50   * call the {@link GeoParser#parse} method on your text string.
51   *
52   */
53  public class GeoParser {
54  	
55  	private static final Logger logger = LoggerFactory.getLogger(GeoParser.class);
56  
57  	// entity extractor to find location names in text
58  	private LocationExtractor locationExtractor;
59  	
60  	// finds coordinates in text.
61  	private CoordinateExtractor coordinateExtractor;
62  	
63  	// resolver to match location names against gazetteer records
64  	private LocationResolver resolver;
65  	
66  	/**
67  	 * Initialize the GeoParse with its dependent interfaces.
68  	 * @param locationExtractor Extract Locations from Text
69  	 * @param coordinateExtractor Extract Coordinates from Text
70  	 * @param resolver Resolve Locations and Coordinates.
71  	 */
72  	public GeoParser(
73  			LocationExtractor locationExtractor, 
74  			CoordinateExtractor coordinateExtractor, 
75  			LocationResolver resolver){
76  		
77  		this.locationExtractor = locationExtractor;
78  		this.coordinateExtractor = coordinateExtractor;
79  		this.resolver = resolver;
80  	}
81  	
82  	/**
83  	 * Takes an unstructured text document (as a String), extracts the
84  	 * location names contained therein, and resolves them into
85  	 * geographic entities representing the best match for those
86  	 * location names.
87  	 * 
88  	 * @param inputText		unstructured text to be processed
89  	 * @return				Locations and Coordinates resolved to Places and where
90  	 * they occurred in the text.
91  	 * @throws Exception
92  	 */
93  	public ResolutionContext parse(String inputText) throws Exception {
94  		
95  		logger.info("Input Size: {}", inputText.length());
96  		
97  		// first, extract location names from the text
98  		List<LocationOccurrence> locationNames = locationExtractor.extractLocationNames(inputText);
99  		
100 		logger.info("Extracted Location Count: {}", locationNames.size());
101 		
102 		// next, extract coordinates from the text
103 		List<CoordinateOccurrence<?>> coordinates = coordinateExtractor.extractCoordinates(inputText);
104 		
105 		logger.info("Extracted Coordinates Count: {}", coordinates.size());
106 		
107 		// build an extraction context
108 		ExtractionContext extractionContext = new ExtractionContext(inputText, locationNames, coordinates);
109 		
110 		// then, resolve the extracted location names and coordinates against a
111 		// gazetteer to produce geographic entities representing the
112 		// locations mentioned in the original text
113 		ResolutionContext resolutionContext = resolver.resolveLocations(extractionContext);
114 		
115 		logger.info("Resolved {} locations and {} coordinates.", 
116 				resolutionContext.getLocations().size(),
117 				resolutionContext.getCoordinates().size());
118 		
119 		return resolutionContext;
120 	}
121 
122 	/**
123 	 * Get the Location Extractor.
124 	 * @return Location Extractor.
125 	 */
126 	public LocationExtractor getLocationExtractor() {
127 		return locationExtractor;
128 	}
129 
130 	/**
131 	 * Get the Coordinate Extractor.
132 	 * @return Coordinate Extractor.
133 	 */
134 	public CoordinateExtractor getCoordinateExtractor() {
135 		return coordinateExtractor;
136 	}
137 
138 	/**
139 	 * Get the Location Resolver.
140 	 * @return Location Resolver.
141 	 */
142 	public LocationResolver getLocationResolver() {
143 		return resolver;
144 	}
