39  /**
40   * If you find some other way to parse a Decimal-degree string that does
41   * not conform to the Regex in {@link DdPatternParsingStrategy}, extend this,
42   * use the named capturing groups provided below in your Regex statement, 
43   * and this class will the rest of the work.
44   */
45  public abstract class BaseDdPatternParsingStrategy
46  		extends BaseRegexPatternParsingStrategy
47  		implements RegexCoordinateParsingStrategy<LatLon>  {
48  
49  	private static final Logger logger = 
50  			LoggerFactory.getLogger(BaseDdPatternParsingStrategy.class);
51  	
52  	/**
53  	 * Parse the matchedString returning a lat/lon occurrence.  This
54  	 * implementation relies on named capture groups:
55  	 * 
56  	 * latdd: Decimal Degrees latitude
57  	 * lathemi: (optional) Hemisphere of latitude
58  	 * londd: Decimal Degrees longitude
59  	 * lonhemi: (optional) Hemisphere of longitude
60  	 * 
61  	 * @param matchedString String representing the lat/lon matched by
62  	 * the Regex statement.
63  	 * @param namedGroups A map of name capture groups with their values.
64  	 * @param startPosition Position where the extraction occurred in text.
65  	 * This is needed for creating an Occurrence object.
66  	 * @return A lat/lon coordinate occurrence.
67  	 */
68  	@Override
69  	public CoordinateOccurrence<LatLon> parse(String matchedString,
70  			Map<String, String> namedGroups, int startPosition) {
71  		
72  		double latitude = parseDecimalDegrees(
73  				namedGroups.get("latdd"),
74  				namedGroups.get("lathemi"));
75  		
76  		double longitude = parseDecimalDegrees(
77  				namedGroups.get("londd"),
78  				namedGroups.get("lonhemi"));
79  		
80  		logger.debug("From '{}', parsed Lat/Lon: {}, {}.", 
81  				new Object[]{ matchedString, latitude, longitude });
82  		
83  		LatLon latlon = new LatLon(latitude, longitude);
84  		
85  		return new LatLonOccurrence(startPosition, matchedString, latlon);
86  	}
87  	
88  	/**
89  	 * Parse the decimal degree representation of a latitude
90  	 * or longitude from a string.
91  	 * 
92  	 * @param decimalDegrees Decimal representation of the coordinate.
93  	 * @param hemisphere Hemisphere of the latitude or longitude.
94  	 * @return Decimal representation of the latitude or longitude.
95  	 */
96  	public double parseDecimalDegrees(String decimalDegrees, String hemisphere){
97  		
98  		int hemi = 1;
99  		
100 		if (hemisphere != null && !hemisphere.trim().isEmpty())
101 			hemi = hemisphere.toUpperCase().equalsIgnoreCase("N")
102 				  || hemisphere.toUpperCase().equalsIgnoreCase("E") ? 1 : -1;
103 		
104 		double dd = tryParse(decimalDegrees, 0.0d);
105 		
106 		return hemi * dd;
107 	}
