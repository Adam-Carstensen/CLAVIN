47  /**
48   * Extracts location names from unstructured text documents using a
49   * named entity recognizer (Apache OpenNLP Name Finder).
50   *
51   */
52  public class ApacheExtractor implements LocationExtractor {
53      
54  	// the actual named entity recognizer (NER) object
55  	private NameFinderME nameFinder;
56  	
57  	// used to tokenize plain text into the OpenNLP format
58  	private TokenizerME tokenizer;
59  
60      // used to split the input into sentences before finding names
61      private SentenceDetectorME sentenceDetector;
62  	
63  	// resource files used by Apache OpenNLP Name Finder
64  	public static InputStream pathToNERModel = 
65  			ApacheExtractor.class.getResourceAsStream("/en-ner-location.bin");
66  	public static InputStream pathToTokenizerModel = 
67  			ApacheExtractor.class.getResourceAsStream("/en-token.bin");
68      public static InputStream pathToSentenceDetectorModel = 
69      			ApacheExtractor.class.getResourceAsStream("/en-sent.bin");

71  	/**
72  	 * Builds an {@link ApacheExtractor} by instantiating the OpenNLP
73  	 * Name Finder and Tokenizer using the default ENGLISH models.
74  	 * 
75  	 * @throws Exception 
76  	 */
77  	public ApacheExtractor() throws Exception {
78  		
79  		this(pathToNERModel, pathToTokenizerModel, pathToSentenceDetectorModel);
80  	}

83  	 * Initialize the Apache OpenNLP extractor with the appropriate models.
84  	 * @param nerModel NER Location Model
85  	 * @param tokenizerModel Word Tokenizer Model
86  	 * @param sentenceModel Sentence Detector Model
87  	 * @throws Exception
88  	 */
89  	public ApacheExtractor(String nerModel, String tokenizerModel, String sentenceModel) throws Exception {
90  		
91  		this(new FileInputStream(nerModel), 
92  			 new FileInputStream(tokenizerModel), 
93  			 new FileInputStream(sentenceModel));
94  	}

96  	/**
97  	 * Instantiate core components of the Extractor from InputStreams.
98  	 * @param nerModel NER Location Model
99  	 * @param tokenizerModel Word Tokenizer Model
100 	 * @param sentenceModel Sentence Detector Model
101 	 * @throws Exception
102 	 */
103 	protected ApacheExtractor(InputStream nerModel, InputStream tokenizerModel, InputStream sentenceModel) throws Exception{
104 		
105 		nameFinder = new NameFinderME(new TokenNameFinderModel(nerModel));
106 		
107 		tokenizer = new TokenizerME(new TokenizerModel(tokenizerModel));
108 		
109         sentenceDetector = new SentenceDetectorME(new SentenceModel(sentenceModel));
110 	}
111 	
112 	/**
113 	 * Extracts location names from unstructured text using the named
114 	 * entity recognizer (NER) feature provided by the Apache OpenNLP
115 	 * Name Finder.
116 	 * 
117 	 * @param plainText		Contents of text document
118      * @return List of location names and positions
119      */
120     public List<LocationOccurrence> extractLocationNames(String plainText) {
121         if(plainText == null) {
122             throw new IllegalArgumentException("plaintext input to extractLocationNames should not be null");
123         }
124 
125         List<LocationOccurrence> nerResults = new ArrayList<LocationOccurrence>();
126 
127         //First step: find the start and end position of each sentence in the document.
128         //each sentence gets processed on its own.
129         //the values used in these Spans are string character offsets
130         Span sentenceSpans[] = sentenceDetector.sentPosDetect(plainText);
131 
132         for (Span sentenceSpan : sentenceSpans) {
133 
134             String sentence = plainText.substring(sentenceSpan.getStart(), sentenceSpan.getEnd());
135 
136             // tokenize the text into the required OpenNLP format
137             String[] tokens = tokenizer.tokenize(sentence);
138 
139             //the values used in these Spans are string character offsets of each token from the sentence beginning
140             Span[] tokenPositionsWithinSentence = tokenizer.tokenizePos(sentence);
141 
142             // find the location names in the tokenized text
143             // the values used in these Spans are NOT string character offsets, they are indices into the 'tokens' array
144             Span names[] = nameFinder.find(tokens);
145 
146 
147             //for each name that got found, create our corresponding occurrence
148             for (Span name : names) {
149 
150                 //find offsets relative to the start of the sentence
151                 int beginningOfFirstWord = tokenPositionsWithinSentence[name.getStart()].getStart();
152                 int endOfLastWord = tokenPositionsWithinSentence[name.getEnd() - 1].getEnd(); //-1 because the high end of a Span is noninclusiv
153 
154                 //to get offsets relative to the document as a whole, just add the offset for the sentence itself
155                 int startOffsetInDoc = sentenceSpan.getStart() + beginningOfFirstWord;
156                 int endOffsetInDoc = sentenceSpan.getStart() + endOfLastWord;
157 
158                 //look back into the original input string to figure out what the text is that I got a hit on
159                 String nameInDocument = plainText.substring(startOffsetInDoc, endOffsetInDoc);
160 
161                 nerResults.add(new LocationOccurrence(nameInDocument, startOffsetInDoc));
162             }
163 
164         }
165 
166         // this is necessary to maintain consistent results across
167         // multiple runs on the same data, which is what we want
168         nameFinder.clearAdaptiveData();
169 
170         return nerResults;
171     }
